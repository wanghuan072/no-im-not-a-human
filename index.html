// 基于CloverLeaf策略的并行化资源加载 function parallelResourceLoading() { if ('IntersectionObserver' in window) { const observer = new IntersectionObserver((entries) => { // 并行处理多个元素 const promises = entries.map(entry => { if (entry.isIntersecting) { const element = entry.target; if (element.dataset.lazyLoad) { return loadResourceAsync(element); } } return Promise.resolve(); }); // 并行执行所有加载任务 Promise.all(promises).then(() => { console.log('Parallel loading completed'); }); }, { rootMargin: '50px', // 提前50px开始加载 threshold: 0.1 }); // 观察所有需要延迟加载的元素 document.querySelectorAll('[data-lazy-load]').forEach(el => { observer.observe(el); }); } } // 异步资源加载函数 async function loadResourceAsync(element) { const resourceType = element.dataset.lazyLoad; const resourceUrl = element.dataset.src; if (resourceType === 'image') { return new Promise((resolve) => { const img = new Image(); img.onload = () => { element.src = resourceUrl; element.classList.add('loaded'); resolve(); }; img.src = resourceUrl; }); } if (resourceType === 'script') { return new Promise((resolve) => { const script = document.createElement('script'); script.src = resourceUrl; script.onload = resolve; document.head.appendChild(script); }); } } // 基于CloverLeaf数据局部性优化的资源预取 function optimizeDataLocality() { // 预取关键资源，提高缓存命中率 const criticalResources = [ '/images/1.webp', '/images/about-img.webp', '/images/logo.webp' ]; criticalResources.forEach(resource => { const link = document.createElement('link'); link.rel = 'prefetch'; link.href = resource; document.head.appendChild(link); }); // 使用requestIdleCallback在浏览器空闲时预加载非关键资源 if ('requestIdleCallback' in window) { requestIdleCallback(() => { const nonCriticalResources = [ '/images/2.webp', '/images/3.webp', '/images/4.webp' ]; nonCriticalResources.forEach(resource => { const link = document.createElement('link'); link.rel = 'prefetch'; link.href = resource; document.head.appendChild(link); }); }); } } // 在DOM加载完成后初始化优化 document.addEventListener('DOMContentLoaded', () => { parallelResourceLoading(); optimizeDataLocality(); });
